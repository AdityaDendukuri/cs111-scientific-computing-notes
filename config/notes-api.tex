% ============================================================
% CS111 Cornell-Style Notes — Macro Library
%
% Layout convention:
%   Left margin  → colored label  (type + number)
%   Body text    → content        (inline term/title + matter)
%
% Public API (all source files should use only these):
%   \addDef{Term}{Content}
%   \addThm{Name}{Content}    — numbered
%   \addLem{Name}{Content}    — numbered (shared counter with Thm)
%   \addCor{Name}{Content}    — numbered (shared counter with Thm)
%   \addProp{Name}{Content}   — numbered (shared counter with Thm)
%   \addPrf{Content}
%   \addNote{Content}
%   \addExmpl{Content}
%   \addExcr{Content}         — numbered
%   \textBox{Content}         — unnumbered expository block
%   \sectionend
% ============================================================

\setlength{\parindent}{0pt}

%% ── Vertical rhythm ──────────────────────────────────────────
\newcommand{\newentry}{\vskip 13pt}

%% ── Internal primitive ───────────────────────────────────────
%% \mLabel{color}{text} is the ONLY place a margin note is written.
%% All add* commands below call this; nothing else should.
\makeatletter
\newcommand{\mLabel}[2]{%
  \reversemarginpar\marginnote{\textcolor{#1}{\textbf{#2}}}%
}
\makeatother

%% ── Counters ─────────────────────────────────────────────────
%% Theorems, Lemmas, Corollaries, and Propositions share one
%% counter (standard math-textbook convention), reset each section.
\newcounter{thmctr}[section]
\renewcommand{\thethmctr}{\thesection.\arabic{thmctr}}

%% Exercises have their own counter, also reset each section.
\newcounter{excrctr}[section]
\renewcommand{\theexcrctr}{\thesection.\arabic{excrctr}}

%% ── Definitions ──────────────────────────────────────────────
%% Margin: "Def" in navy.  Inline: bold term followed by content.
\newcommand{\addDef}[2]{%
  \newentry
  \mLabel{ucsb_blue}{Def}%
  \textbf{#1.}\enspace #2%
}

%% ── Theorems ─────────────────────────────────────────────────
\newcommand{\addThm}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \mLabel{ucsb_blue}{Thm~\thethmctr}%
  \textit{Theorem~\thethmctr} \textbf{(#1).}\enspace #2%
}

%% ── Lemmas ───────────────────────────────────────────────────
\newcommand{\addLem}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \mLabel{ucsb_blue}{Lem~\thethmctr}%
  \textit{Lemma~\thethmctr} \textbf{(#1).}\enspace #2%
}

%% ── Corollaries ──────────────────────────────────────────────
%% Green label to visually distinguish "follows from" statements.
\newcommand{\addCor}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \mLabel{ucsb_green}{Cor~\thethmctr}%
  \textit{Corollary~\thethmctr} \textbf{(#1).}\enspace #2%
}

%% ── Propositions ─────────────────────────────────────────────
\newcommand{\addProp}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \mLabel{ucsb_blue}{Prop~\thethmctr}%
  \textit{Proposition~\thethmctr} \textbf{(#1).}\enspace #2%
}

%% ── Proofs ───────────────────────────────────────────────────
%% Opens with italic "Proof." and closes with □.
\newcommand{\addPrf}[1]{%
  \newentry
  \mLabel{ucsb_green}{Proof}%
  \textit{Proof.}\enspace #1\hfill$\square$%
}

%% ── Notes ────────────────────────────────────────────────────
\newcommand{\addNote}[1]{%
  \newentry
  \mLabel{ucsb_green}{Note}%
  #1%
}

%% ── Examples ─────────────────────────────────────────────────
\newcommand{\addExmpl}[1]{%
  \newentry
  \mLabel{ucsb_gold}{Example}%
  #1%
}

%% ── Exercises ────────────────────────────────────────────────
%% Numbered per section.  \label/\ref works on these.
\newcommand{\addExcr}[1]{%
  \refstepcounter{excrctr}%
  \newentry
  \mLabel{ucsb_coral}{Ex~\theexcrctr}%
  #1%
}

%% ── Expository blocks (no margin label) ──────────────────────
%% For derivations, extended calculations, and narrative text
%% that sits between labeled entries.
\newcommand{\textBox}[1]{%
  \newentry
  #1%
}

%% ── Section-end marker ───────────────────────────────────────
\newcommand{\sectionend}{%
  \mLabel{ucsb_coral}{\S~end}%
}
