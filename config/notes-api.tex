% ============================================================
% CS111 Cornell-Style Notes — Macro Library
%
% Layout convention:
%   Left margin  → colored label  (type + number)
%   Body text    → content        (inline term/title + matter)
%
% Labels use \llap so they extend leftward into the margin as a
% zero-width box.  Body text therefore always starts at the same
% \textwidth edge regardless of label length — no marginnote
% drift, no marginparwidth overflow.
%
% Requires: left margin ≥ ~30mm  (set once in import-packages).
%
% Public API (all source files should use only these):
%   \addDef{Term}{Content}    — numbered (own counter)
%   \addThm{Name}{Content}    — numbered (shared counter with Lem/Cor/Prop)
%   \addLem{Name}{Content}    — numbered (shared counter with Thm)
%   \addCor{Name}{Content}    — numbered (shared counter with Thm)
%   \addProp{Name}{Content}   — numbered (shared counter with Thm)
%   \addPrf{Content}
%   \addNote{Content}         — numbered (own counter)
%   \addExmpl{Content}
%   \addExcr{Content}         — numbered (own counter)
%   \textBox{Content}         — unnumbered expository block
%   \sectionend
% ============================================================

\setlength{\parindent}{0pt}

%% ── Vertical rhythm ──────────────────────────────────────────
\newcommand{\newentry}{\vskip 13pt}

%% ── Internal primitive ───────────────────────────────────────
%% \mLabel{color}{text}
%%   \llap places a zero-width right-aligned box at the start of
%%   the line; its content extends LEFT into the margin.
%%   The 6mm \hspace creates a consistent gap between label and text.
\newcommand{\mLabel}[2]{%
  \llap{\textcolor{#1}{\textbf{#2}}\hspace{6mm}}%
}

%% ── Counters ─────────────────────────────────────────────────
%% Thm/Lem/Cor/Prop share one counter; Def and Note have their own.
\newcounter{thmctr}[section]
\renewcommand{\thethmctr}{\thesection.\arabic{thmctr}}

\newcounter{defctr}[section]
\renewcommand{\thedefctr}{\thesection.\arabic{defctr}}

\newcounter{notectr}[section]
\renewcommand{\thenotectr}{\thesection.\arabic{notectr}}

\newcounter{excrctr}[section]
\renewcommand{\theexcrctr}{\thesection.\arabic{excrctr}}

%% ── Definitions ──────────────────────────────────────────────
\newcommand{\addDef}[2]{%
  \refstepcounter{defctr}%
  \newentry
  \noindent\mLabel{ucsb_blue}{Def~\thedefctr}%
  \textbf{(#1)}\enspace #2%
}

%% ── Theorems ─────────────────────────────────────────────────
\newcommand{\addThm}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \noindent\mLabel{ucsb_blue}{Thm~\thethmctr}%
  \textbf{(#1)}\enspace #2%
}

%% ── Lemmas ───────────────────────────────────────────────────
\newcommand{\addLem}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \noindent\mLabel{ucsb_blue}{Lem~\thethmctr}%
  \textbf{(#1)}\enspace #2%
}

%% ── Corollaries ──────────────────────────────────────────────
\newcommand{\addCor}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \noindent\mLabel{ucsb_green}{Cor~\thethmctr}%
  \textbf{(#1)}\enspace #2%
}

%% ── Propositions ─────────────────────────────────────────────
\newcommand{\addProp}[2]{%
  \refstepcounter{thmctr}%
  \newentry
  \noindent\mLabel{ucsb_blue}{Prop~\thethmctr}%
  \textbf{(#1)}\enspace #2%
}

%% ── Proofs ───────────────────────────────────────────────────
\newcommand{\addPrf}[1]{%
  \newentry
  \noindent\mLabel{ucsb_green}{Proof}%
  \textit{Proof.}\enspace #1\hfill$\square$%
}

%% ── Notes ────────────────────────────────────────────────────
\newcommand{\addNote}[1]{%
  \refstepcounter{notectr}%
  \newentry
  \noindent\mLabel{ucsb_green}{Note~\thenotectr}%
  #1%
}

%% ── Examples ─────────────────────────────────────────────────
\newcommand{\addExmpl}[1]{%
  \newentry
  \noindent\mLabel{ucsb_gold}{Example}%
  #1%
}

%% ── Exercises ────────────────────────────────────────────────
\newcommand{\addExcr}[1]{%
  \refstepcounter{excrctr}%
  \newentry
  \noindent\mLabel{ucsb_coral}{Ex~\theexcrctr}%
  #1%
}

%% ── Expository blocks (no margin label) ──────────────────────
\newcommand{\textBox}[1]{%
  \newentry
  #1%
}

%% ── Section-end marker ───────────────────────────────────────
\newcommand{\sectionend}{%
  \noindent\mLabel{ucsb_coral}{\S~end}%
}
