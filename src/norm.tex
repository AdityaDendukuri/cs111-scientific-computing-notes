\subsection{Vector and Matrix Norms}
\label{sub:norm-theory}

Foundations for measuring approximation errors and convergence.

\subsubsection{Vector Norms}

\addDef{Vector Norm}{%
$\|\cdot\| : \fR^n \to \fR$ satisfies:
\begin{enumerate}
    \item \textbf{Definiteness:}\enspace $\|\bx\| \geq 0$, with equality iff $\bx = \bzero$.
    \item \textbf{Homogeneity:}\enspace $\|c\bx\| = |c|\|\bx\|$.
    \item \textbf{Triangle Inequality:}\enspace $\|\bx + \by\| \leq \|\bx\| + \|\by\|$.
\end{enumerate}
}
\label{def:vector-norm}

\addDef{$\ell^p$ Norms}{%
$\|\bx\|_p = (\sum |x_i|^p)^{1/p}$.
\begin{itemize}
    \item \textbf{$\ell_1$:}\enspace $\sum |x_i|$ (Sparsity).
    \item \textbf{$\ell_2$:}\enspace $\sqrt{\bx^T\bx}$ (Energy).
    \item \textbf{$\ell_\infty$:}\enspace $\max |x_i|$ (Tolerance).
\end{itemize}
}
\label{def:lp-norms}

\addThm{Fundamental Inequalities}{%
\begin{itemize}
    \item \textbf{Cauchy-Schwarz:}\enspace $|\bx^T\by| \leq \|\bx\|_2 \|\by\|_2$.
    \item \textbf{HÃ¶lder:}\enspace $|\bx^T\by| \leq \|\bx\|_p \|\by\|_q$ where $1/p + 1/q = 1$.
\end{itemize}
}
\label{thm:norm-inequalities}

\addNote{%
\textbf{Equivalence:}\enspace
On $\fR^n$, all norms are equivalent: $c_1 \|\bx\|_\beta \leq \|\bx\|_\alpha \leq c_2 \|\bx\|_\beta$. Convergence in one implies convergence in all.
}

\addExcr{%
\begin{enumerate}
    \item Prove the triangle inequality for $\|\cdot\|_2$ using Cauchy-Schwarz.
    \item Show that $\|\bx\|_\infty = \lim_{p\to\infty} \|\bx\|_p$.
\end{enumerate}
}

\subsubsection{Matrix Norms}

Measure how much a matrix amplifies vectors.

\addDef{Induced (Operator) Norm}{%
$\|\bA\| = \max_{\bx \neq \bzero} \frac{\|\bA\bx\|}{\|\bx\|}$.
\begin{itemize}
    \item \textbf{$\|\bA\|_1$:}\enspace Max absolute column sum.
    \item \textbf{$\|\bA\|_\infty$:}\enspace Max absolute row sum.
    \item \textbf{$\|\bA\|_2$ (Spectral):}\enspace $\sigma_{\max}(\bA) = \sqrt{\lambda_{\max}(\bA^T\bA)}$.
\end{itemize}
}
\label{def:matrix-norm}

\addDef{Frobenius Norm}{%
$\|\bA\|_F = \sqrt{\sum a_{ij}^2} = \sqrt{\text{Tr}(\bA^T\bA)}$.
\begin{itemize}
    \item \textbf{Relation:}\enspace $\|\bA\|_2 \leq \|\bA\|_F \leq \sqrt{\text{rank}(\bA)} \|\bA\|_2$.
\end{itemize}
}
\label{def:frobenius-norm}

\addNote{%
\textbf{Consistency and Submultiplicativity:}\enspace
Matrix norms are useful when they satisfy $\|\bA\bx\| \leq \|\bA\|\|\bx\|$. All induced norms are additionally submultiplicative: $\|\bA\bB\| \leq \|\bA\|\|\bB\|$.
}

\addExcr{%
\begin{enumerate}
    \item Compute $\|\bA\|_1, \|\bA\|_\infty, \|\bA\|_2, \|\bA\|_F$ for $\bA = \begin{pmatrix} 1 & -2 \\ -3 & 4 \end{pmatrix}$.
    \item \textbf{Low-Rank Approximation:}\enspace The Frobenius norm is central to the Eckart-Young-Mirsky theorem. How is it related to singular values?
    \item Find a $2 \times 2$ example where the max norm ($\max|a_{ij}|$) is not submultiplicative.
\end{enumerate}
}
