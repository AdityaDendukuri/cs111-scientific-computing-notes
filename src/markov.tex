\subsection{Markov Chains and Stochastic Matrices}
\label{sub:markov}

Modeling random processes where future state depends only on the current state.

\subsubsection{Definitions}

\addDef{Column-Stochastic Matrix}{%
Matrix $\bP \in \fR^{n \times n}$ where:
\begin{enumerate}
    \item $P_{ij} \geq 0$.
    \item $\sum_i P_{ij} = 1$ (Columns sum to 1).
\end{enumerate}
\textbf{Property:}\enspace $\lambda=1$ is always an eigenvalue (dominant).
}

\addDef{Markov Chain}{%
Probability distributions $\bx^{(k)}$ evolving by $\bx^{(k+1)} = \bP \bx^{(k)}$.
\begin{itemize}
    \item After $k$ steps: $\bx^{(k)} = \bP^k \bx^{(0)}$.
    \item Long-run behavior is governed by the eigendecomposition of $\bP$.
\end{itemize}
}

\addDef{Stationary Distribution}{%
A probability vector $\boldsymbol{\pi}$ such that **$\bP \boldsymbol{\pi} = \boldsymbol{\pi}$**.
\begin{itemize}
    \item Right eigenvector of $\bP$ for $\lambda=1$.
    \item Represents the equilibrium state of the chain.
\end{itemize}
}

\subsubsection{Convergence and Mixing}

\addThm{Perron-Frobenius for Chains}{%
If $\bP$ is \textbf{irreducible} (all states reachable) and \textbf{primitive} (no cycles), then:
\begin{enumerate}
    \item $\lambda_1 = 1$ is simple and unique.
    \item $|\lambda_i| < 1$ for all other eigenvalues.
    \item $\lim_{k\to\infty} \bP^k \bx^{(0)} = \boldsymbol{\pi}$ for any initial state.
\end{enumerate}
}

\addDef{Spectral Gap and Mixing Time}{%
The gap **$\delta = 1 - |\lambda_2|$** controls convergence speed.
\begin{itemize}
    \item \textbf{Mixing Time:}\enspace $k_{\varepsilon} \approx \log(1/\varepsilon) / \delta$.
    \item \textbf{Insight:}\enspace Large gap = fast mixing. Connected graphs have larger gaps.
\end{itemize}
}

\subsubsection{Properties and Design}

\addDef{Detailed Balance (Reversibility)}{%
A chain is reversible if $\pi_j P_{ij} = \pi_i P_{ji}$ for all $i, j$.
\begin{itemize}
    \item \textbf{Implication:}\enspace Reversible chains have **real eigenvalues**.
    \item \textbf{Symmetry:}\enspace $\bD_{\boldsymbol{\pi}}^{-1/2} \bP \bD_{\boldsymbol{\pi}}^{1/2}$ is symmetric.
\end{itemize}
}

\addNote{%
\textbf{Design Tool: Metropolis-Hastings.}\enspace Construct a reversible chain for a desired $\boldsymbol{\pi}$ by accepting moves with probability $a = \min(1, \frac{\pi_{\text{new}} Q(\text{old}|\text{new})}{\pi_{\text{old}} Q(\text{new}|\text{old})})$.
}
