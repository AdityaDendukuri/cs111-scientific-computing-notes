\subsection{Eigenvalues and Eigenvectors}
\label{sub:eigen-theory}

Reveals the scaling directions of a linear transformation. Essential for stability analysis, PCA, and spectral methods.

\subsubsection{Definitions and Basic Properties}

\addDef{Eigenvalues and Eigenvectors}{%
For $\bA \in \fR^{n \times n}$, a scalar $\lambda \in \fC$ is an \textbf{eigenvalue} if $\exists$ nonzero $\bv \in \fC^n$ such that:
\begin{align*}
    \bA\bv = \lambda\bv.
\end{align*}
\begin{itemize}
    \item \textbf{Spectrum $\sigma(\bA)$:}\enspace The set of all eigenvalues.
    \item \textbf{Invariance:}\enspace Multiplying $\bv$ by $\bA$ only scales its length; its direction remains fixed.
\end{itemize}
}
\label{def:eigenvalue}

\addDef{Characteristic Polynomial}{%
$p_{\bA}(\lambda) = \det(\bA - \lambda \bI)$. Eigenvalues are the roots of $p_{\bA}(\lambda) = 0$.
}
\label{def:char-poly}

\addExcr{%
\begin{enumerate}
    \item Verify $\bv = (1, 1)^T$ is an eigenvector of $\bA = \begin{pmatrix}3&1\\1&3\end{pmatrix}$ and find $\lambda$.
    \item Why is the zero vector excluded from the definition of an eigenvector?
\end{enumerate}
}

\addThm{Similar Matrices}{%
If $\bB = \bV^{-1}\bA\bV$ (Similarity Transform), then $\sigma(\bB) = \sigma(\bA)$.
\textbf{Insight:}\enspace Eigenvalues are invariant under change of basis.
}
\label{thm:similar-matrices}

\addThm{Key Properties}{%
\begin{enumerate}
    \item \textbf{Trace:}\enspace $\sum \lambda_i = \text{tr}(\bA)$.
    \item \textbf{Determinant:}\enspace $\prod \lambda_i = \det(\bA)$.
    \item \textbf{Singularity:}\enspace $0 \in \sigma(\bA) \iff \bA$ is singular.
    \item \textbf{Inversion:}\enspace If $\bA$ is invertible, eigenvalues of $\bA^{-1}$ are $1/\lambda_i$.
\end{enumerate}
}
\label{thm:eigen-props}

\subsubsection{Symmetric and Hermitian Matrices}

The ``Golden State'' of eigenvalue problems.

\addThm{Spectral Theorem (Real Symmetric)}{%
If $\bA = \bA^T$:
\begin{enumerate}
    \item All eigenvalues $\lambda_i$ are \textbf{real}.
    \item Eigenvectors for distinct $\lambda$ are \textbf{orthogonal}.
    \item $\bA$ is \textbf{orthogonally diagonalizable}: $\bA = \bQ\bD\bQ^T$.
\end{enumerate}
}
\label{thm:spectral-symmetric}

\addDef{Hermitian Matrix}{%
$\bA = \bA^*$ (conjugate transpose). The complex analog of symmetric matrices.
\begin{itemize}
    \item \textbf{Spectral Theorem (Hermitian):}\enspace Unitarily diagonalizable ($\bA = \bQ\bD\bQ^*$) with real eigenvalues.
\end{itemize}
}
\label{def:hermitian}

\addExcr{%
\begin{enumerate}
    \item Prove eigenvalues of $\bA^2$ are $\lambda_i^2$.
    \item Diagonalize $\bA = \begin{pmatrix}4&1\\1&4\end{pmatrix}$ by finding an orthonormal eigenbasis.
\end{enumerate}
}

\subsubsection{Iterative Algorithms}

In practice, we compute eigenvalues via iteration, not by finding roots of $p_\bA(\lambda)$.

\addDef{Power Iteration}{%
Repeatedly apply $\bA$ to a unit vector: $\bv^{(k)} = \frac{\bA\bv^{(k-1)}}{\|\bA\bv^{(k-1)}\|}$.
\begin{itemize}
    \item \textbf{Convergence:}\enspace $\bv^{(k)} \to$ dominant eigenvector (for $|\lambda_1| > |\lambda_2|$).
    \item \textbf{Rate:}\enspace $O(|\lambda_2/\lambda_1|^k)$.
\end{itemize}
}
\label{def:power-iteration}

\addPrf{%
Assume $\bA$ is diagonalizable with eigenvectors $\{\bu_1, \dots, \bu_n\}$ and eigenvalues $|\lambda_1| > |\lambda_2| \geq \dots$. Let $\bx^{(0)} = \sum_{i=1}^n c_i \bu_i$ with $c_1 \neq 0$. Applying $\bA$ $k$ times:
\begin{align*}
    \bA^k \bx^{(0)} &= \sum_{i=1}^n c_i \lambda_i^k \bu_i \\
    &= c_1 \lambda_1^k \left( \bu_1 + \sum_{i=2}^n \frac{c_i}{c_1} \left(\frac{\lambda_i}{\lambda_1}\right)^k \bu_i \right).
\end{align*}
Since $|\lambda_i/\lambda_1| < 1$ for all $i > 1$, the summation term decays to zero as $k \to \infty$. Thus $\bA^k \bx^{(0)}$ aligns with $\bu_1$. The normalization at each step prevents overflow while preserving this alignment.
}

\addDef{Inverse Iteration}{%
Apply power iteration to $(\bA - \sigma\bI)^{-1}$.
\begin{itemize}
    \item \textbf{Use:}\enspace Finds the eigenvalue closest to the shift $\sigma$.
    \item \textbf{Efficiency:}\enspace Factorize $(\bA - \sigma\bI)$ once, then solve at each step ($O(n^2)$).
\end{itemize}
}
\label{def:inverse-iteration}

\addDef{Rayleigh Quotient Iteration (RQI)}{%
Update the shift $\sigma_k$ to the current Rayleigh quotient $(\bv^{(k)})^T\bA\bv^{(k)}$ at each step.
\begin{itemize}
    \item \textbf{Convergence:}\enspace \textbf{Cubic} for symmetric matrices (digits triple each step).
\end{itemize}
}
\label{def:rqi}

\addExcr{%
\begin{enumerate}
    \item Implement power iteration to find $\lambda_{\max}$ of a tridiagonal matrix.
    \item Use RQI on a random $4 \times 4$ symmetric matrix. Observe the rate of convergence.
\end{enumerate}
}

\subsubsection{Schur Decomposition and the QR Algorithm}

The theoretical and practical backbone of general eigensolvers.

\addThm{Schur Decomposition}{%
Every square $\bA$ satisfies $\bA = \bQ\bT\bQ^*$, where $\bQ$ is unitary and $\bT$ is upper triangular.
\begin{itemize}
    \item \textbf{Schur Vectors:}\enspace Columns of $\bQ$ provide an orthonormal basis for nested invariant subspaces.
\end{itemize}
}
\label{thm:schur}

\addDef{QR Iteration}{%
Iteratively compute QR factorizations and swap the factors:
\begin{enumerate}
    \item $\bA_k = \bQ_k \bR_k$
    \item $\bA_{k+1} = \bR_k \bQ_k = \bQ_k^T \bA_k \bQ_k$
\end{enumerate}
\textbf{Convergence:}\enspace $\bA_k$ converges to the Schur form $\bT$.
}
\label{def:qr-iteration}

\addNote{%
\textbf{Performance:}\enspace
Raw QR iteration is $O(n^3)$ per step. Practical implementations (like \texttt{np.linalg.eig}) use:
\begin{enumerate}
    \item \textbf{Hessenberg Reduction:}\enspace Pre-process $\bA$ to upper Hessenberg form ($O(n^3)$ once).
    \item \textbf{Triangularization:}\enspace Hessenberg QR steps cost only $O(n^2)$.
    \item \textbf{Shifts:}\enspace Accelerate convergence to quadratic or cubic.
\end{enumerate}
}

\addExcr{%
\begin{enumerate}
    \item For $\bA = \begin{pmatrix}3&1&0\\0&2&1\\0&0&2\end{pmatrix}$, identify the Schur vectors and eigenvalues by inspection.
    \item Implement 50 steps of basic QR iteration on a random $3 \times 3$ symmetric matrix. Check the off-diagonals.
\end{enumerate}
}
