\subsection{Numerical Methods for ODEs}
\label{subsec:ode-methods}

Numerical integration of the \emph{initial value problem} (IVP):
\[
    \frac{dy}{dt} = f(t, y), \qquad y(t_0) = y_0.
\]

\subsubsection{Basic Methods}

\addDef{Forward Euler (Explicit)}{%
Advance via forward difference:
\[
    y_{n+1} = y_n + h f(t_n, y_n).
\]
\textbf{Error:}\enspace Local $O(h^2)$, Global $O(h)$.
\textbf{Stability:}\enspace Unstable for stiff problems. Absolute stability requires $h \lambda \in$ region of stability (circle in left half-plane).
}

\addDef{Backward Euler (Implicit)}{%
Advance via backward difference:
\[
    y_{n+1} = y_n + h f(t_{n+1}, y_{n+1}).
\]
\textbf{Rule:}\enspace Requires solving a nonlinear system at each step (via Newton).
\textbf{Stability:}\enspace **A-Stable** (unconditionally stable for all $h$ if $\text{Re}(\lambda) < 0$). The choice for stiff systems.
}

\subsubsection{Runge-Kutta Methods}

Multistage methods that match higher-order Taylor expansions without requiring analytic derivatives of $f$.

\addDef{RK4 (Classical 4th Order)}{%
Evaluates $f$ at four stages per step:
\begin{align*}
    k_1 &= f(t_n, y_n) \\
    k_2 &= f(t_n + h/2, y_n + h k_1 / 2) \\
    k_3 &= f(t_n + h/2, y_n + h k_2 / 2) \\
    k_4 &= f(t_n + h, y_n + h k_3) \\
    y_{n+1} &= y_n + \frac{h}{6}(k_1 + 2k_2 + 2k_3 + k_4)
\end{align*}
\textbf{Error:}\enspace Local $O(h^5)$, Global **$O(h^4)$**. Standard default for non-stiff systems.
}

\addThm{Adaptive Step Control (RK45)}{%
Uses an **embedded pair** to estimate local error $e$ without extra stages (Dormand-Prince).
\textbf{Mechanism:}\enspace $h_{\text{new}} = h_{\text{old}} (\varepsilon / \|e\|)^{1/5}$.
\textbf{Rule:}\enspace Automatically shrinks $h$ in regions of high curvature and expands $h$ where $f$ is smooth.
}

\subsubsection{Symplectic Integrators}

Preserve the geometric structure of **Hamiltonian systems** (Energy conservation).

\addDef{Velocity Verlet}{%
For $H = \frac{1}{2}v^2 + V(q)$, ensures no long-term energy drift:
\begin{enumerate}
    \item $v_{n+1/2} = v_n + \frac{h}{2} F_n$
    \item $q_{n+1} = q_n + h v_{n+1/2}$
    \item $v_{n+1} = v_{n+1/2} + \frac{h}{2} F_{n+1}$
\end{enumerate}
\textbf{Property:}\enspace Symplectic and time-reversible. Numerical energy oscillates but stays bounded.
}

\addNote{%
\textbf{The Drift Warning:}\enspace Standard explicit methods like Forward Euler or RK4 exhibit monotonic energy growth/decay in Hamiltonian systems. Always use symplectic methods for long-term physics simulations.
}

\subsubsection{Exercises}

\addExcr{%
\begin{enumerate}
    \item Solve $y' = -y, y(0)=1$ with Forward Euler for 4 steps. Compare error to $e^{-1}$.
    \item Verify the stability limit of Forward Euler for $y' = \lambda y$ is $h < 2/|\lambda|$.
    \item Show that Forward Euler on a harmonic oscillator leads to unbounded energy growth.
    \item Implement RK45 for $y' = \cos(t^2)$ and plot the step-size $h$ over time.
\end{enumerate}
}
